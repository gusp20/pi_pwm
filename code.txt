import RPi.GPIO as GPIO
import time
GPIO.setwarnings(False)
GPIO.setmode(GPIO.BCM)
GPIO.setup(21, GPIO.OUT)

led = GPIO.PWM(21, 500)
led.start(0)

while(1):
    for dc in range (0, 101, 5):
        led.ChangeDutyCycle(dc)
        time.sleep(0.1)
       
    for dc in range(100, -1, -5):
        led.ChangeDutyCycle(dc)
        time.sleep(0.1)
